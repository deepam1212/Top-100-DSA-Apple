**Given a string s, return the longest palindromic substring in s.**


 https://leetcode.com/problems/longest-palindromic-substring/description/

 
```swift
Example 1:

Input: s = "babad"
Output: "bab"
Explanation: "aba" is also a valid answer.
Example 2:

Input: s = "cbbd"
Output: "bb"
 ```

**Constraints:**
```swift
1 <= s.length <= 1000
s consist of only digits and English letters.
```

**Solution**
```swift
class Solution {
    func longestPalindrome(_ s: String) -> String {
        //
        var charArr: [Character] = Array(s)
        var outputString: String = ""
        var totalLength: Int = 0
        for (index, item) in charArr.enumerated() {
                //
                var left: Int = index
                var right: Int = index
                while(left >= 0 && right < charArr.count && charArr[left] == charArr[right]) {
                    if (right - left + 1) > totalLength {
                        outputString = String(charArr[left...right])
                        totalLength = right - left + 1
                    }
                        left -= 1
                        right += 1
                }

                left = index
                right = index + 1
                while(left >= 0 && right < charArr.count && charArr[left] == charArr[right]) {
                    if (right - left + 1) > totalLength {
                        outputString = String(charArr[left...right])
                        totalLength = right - left + 1
                    }
                        left -= 1
                        right += 1
                }
        }
        //
        return outputString
    }
}
```

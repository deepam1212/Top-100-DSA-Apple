```swift
Given an array nums with n objects colored red, white, or blue, sort them in-place so that objects of the same color are adjacent, with the colors in the order red, white, and blue.
We will use the integers 0, 1, and 2 to represent the color red, white, and blue, respectively.
You must solve this problem without using the library's sort function.
```

 https://leetcode.com/problems/sort-colors/description/
```swift
Example 1:

Input: nums = [2,0,2,1,1,0]
Output: [0,0,1,1,2,2]
Example 2:

Input: nums = [2,0,1]
Output: [0,1,2]
```

**Constraints:**

```swift
n == nums.length
1 <= n <= 300
nums[i] is either 0, 1, or 2.
```

**Follow up: Could you come up with a one-pass algorithm using only constant extra space?**
```swift
class Solution {
    func sortColors(_ nums: inout [Int]) {
        //
        var count0: Int = 0, count1: Int = 0, count2: Int = 0
        //
        for item in nums {
            if item == 0 {
                count0 += 1
            } else if item == 1 {
                count1 += 1
            } else {
                count2 += 1
            }
        }
        //
        for index in 0..<count0 {
            nums[index] = 0
        }
        //
        for index in count0..<(count0 + count1) {
            nums[index] = 1
        }
        //
        for index in (count0 + count1)..<(count0 + count1 + count2) {
            nums[index] = 2
        }
    }
}
```

**Optimal approach: Dutch National Flag algorithm**

```swift
class Solution {
    func sortColors(_ nums: inout [Int]) {
        //
        var low: Int = 0
        var mid: Int = 0
        var high: Int = nums.count - 1// 0 to low - 1 = 0; low to mid - 1 = 1, mid to high are unknown or filled with unsorted arr; high to n - 1 are 2
        while(mid <= high) {
            if nums[mid] == 0 {
                let temp = nums[mid]
                nums[mid] = nums[low]
                nums[low] = temp
                low += 1
                mid += 1
            } else if nums[mid] == 1 {
                mid += 1
            } else {
                let temp = nums[mid]
                nums[mid] = nums[high]
                nums[high] = temp
                high -= 1
            }
        }
    }
}
```

```swift
Given strings s1, s2, and s3, find whether s3 is formed by an interleaving of s1 and s2.

An interleaving of two strings s and t is a configuration where s and t are divided into n and m substrings respectively, such that:

s = s1 + s2 + ... + sn
t = t1 + t2 + ... + tm
|n - m| <= 1
The interleaving is s1 + t1 + s2 + t2 + s3 + t3 + ... or t1 + s1 + t2 + s2 + t3 + s3 + ...
```
**Note: a + b is the concatenation of strings a and b.**


![Example](assets/interleave.jpg)



**Example 1:**


```swift
Input: s1 = "aabcc", s2 = "dbbca", s3 = "aadbbcbcac"
Output: true
Explanation: One way to obtain s3 is:
Split s1 into s1 = "aa" + "bc" + "c", and s2 into s2 = "dbbc" + "a".
Interleaving the two splits, we get "aa" + "dbbc" + "bc" + "a" + "c" = "aadbbcbcac".
Since s3 can be obtained by interleaving s1 and s2, we return true.
Example 2:

Input: s1 = "aabcc", s2 = "dbbca", s3 = "aadbbbaccc"
Output: false
Explanation: Notice how it is impossible to interleave s2 with any other string to obtain s3.
Example 3:

Input: s1 = "", s2 = "", s3 = ""
Output: true
```

**Constraints:**


```swift
0 <= s1.length, s2.length <= 100
0 <= s3.length <= 200
s1, s2, and s3 consist of lowercase English letters.
```

**Follow up: Could you solve it using only O(s2.length) additional memory space?**

**Solution**

```swift
class Solution {
    func isInterleave(_ s1: String, _ s2: String, _ s3: String) -> Bool {
        
        if s1.count + s2.count != s3.count { return false }
        
        let s1Arr = Array(s1)
        let s2Arr = Array(s2)
        let s3Arr = Array(s3)
        
        var dp = Array(
            repeating: Array(repeating: -1, count: s2Arr.count + 1),
            count: s1Arr.count + 1
        )
        
        return solve(0, 0, s1Arr, s2Arr, s3Arr, &dp) == 1
    }
    
    func solve(_ i: Int,
               _ j: Int,
               _ s1: [Character],
               _ s2: [Character],
               _ s3: [Character],
               _ dp: inout [[Int]]) -> Int {
        
        if i == s1.count && j == s2.count {
            return 1
        }
        
        if dp[i][j] != -1 {
            return dp[i][j]
        }
        
        if i < s1.count,
           s1[i] == s3[i + j],
           solve(i + 1, j, s1, s2, s3, &dp) == 1 {
            dp[i][j] = 1
            return 1
        }
        
        if j < s2.count,
           s2[j] == s3[i + j],
           solve(i, j + 1, s1, s2, s3, &dp) == 1 {
            dp[i][j] = 1
            return 1
        }
        
        dp[i][j] = 0
        return 0
    }
}
```

**Given a binary tree, determine if it is height-balanced.**
```swift
A height-balanced binary tree is a binary tree in which the depth of the two subtrees of every node never differs by more than one.
```
**Example 1:**

![Trapping Rain Water Example](assets/balance_1.jpg)

```swift
Input: root = [3,9,20,null,null,15,7]
Output: true
```

**Example 2:**

![Trapping Rain Water Example](assets/balance_2.jpg)

```swift
Input: root = [1,2,2,3,3,null,null,4,4]
Output: false
```
**Example 3:**
```swift
Input: root = []
Output: true
```
**Solution**
```swift
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     public var val: Int
 *     public var left: TreeNode?
 *     public var right: TreeNode?
 *     public init() { self.val = 0; self.left = nil; self.right = nil; }
 *     public init(_ val: Int) { self.val = val; self.left = nil; self.right = nil; }
 *     public init(_ val: Int, _ left: TreeNode?, _ right: TreeNode?) {
 *         self.val = val
 *         self.left = left
 *         self.right = right
 *     }
 * }
 */
class Solution {
    func isBalanced(_ root: TreeNode?) -> Bool {
        return height(root: root) != -1
    }
    
    func height(root: TreeNode?) -> Int {
        guard let rootNode = root else {return 0}
        //
        let leftHeight = height(root: rootNode.left)
        if leftHeight == -1 {
            return -1
        }
        let rightHeight = height(root: rootNode.right)
        if rightHeight == -1 {
            return -1
        }
        if abs(leftHeight - rightHeight) > 1 {
            return -1
        }
        return max(leftHeight, rightHeight) + 1
    }
}
```
